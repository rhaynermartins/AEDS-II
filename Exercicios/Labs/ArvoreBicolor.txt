//atividade 1 
//atributo cor 
class No {
    int elemento;
    boolean cor;
    No esq;
    No dir;

    No(int elemento) {
        this.elemento = elemento;
        this.cor = true;
        this.esq = null;
        this.dir = null;
    }

    No(int elemento, boolean cor) {
        this.elemento = elemento;
        this.cor = cor;
        this.esq = null;
        this.dir = null;
    }
}

//atividade 2
//detectar 4.no
public boolean isNoTipo4() {
    return esq != null && dir != null && esq.cor && dir.cor;
}

//atividade 3
//recolorir no 
private void fragmentar(No i) {
    i.cor = true;

    if (i.esq != null) {
        i.esq.cor = false;
    }

    if (i.dir != null) {
        i.dir.cor = false;
    }

    if (i == raiz) {
        i.cor = false;
    }
}
